```python
# backend/app/modules/plan/schemas.py
from pydantic import BaseModel, Field
from datetime import datetime
from uuid import UUID
from typing import Optional, List, Dict, Any


# ============ 计划-场景关联 Schemas ============

class PlanScenarioCreate(BaseModel):
    """创建计划-场景关联请求"""
    scenario_id: str
    dataset_id: Optional[str] = None
    variables_override: Optional[Dict[str, Any]] = None
    sort_order: int = 0


class PlanScenarioUpdate(BaseModel):
    """更新计划-场景关联请求"""
    dataset_id: Optional[str] = None
    variables_override: Optional[Dict[str, Any]] = None
    sort_order: Optional[int] = None


class PlanScenarioResponse(BaseModel):
    """计划-场景关联响应"""
    id: UUID
    plan_id: UUID
    scenario_id: UUID
    dataset_id: Optional[UUID]
    variables_override: Dict[str, Any]
    sort_order: int
    created_at: datetime

    # 关联信息
    scenario_name: Optional[str] = None
    dataset_name: Optional[str] = None

    class Config:
        from_attributes = True


# ============ 测试计划 Schemas ============

class TestPlanCreate(BaseModel):
    """创建测试计划请求"""
    name: str = Field(..., min_length=1, max_length=255)
    description: Optional[str] = None
    scenarios: Optional[List[PlanScenarioCreate]] = None


class TestPlanUpdate(BaseModel):
    """更新测试计划请求"""
    name: Optional[str] = Field(None, min_length=1, max_length=255)
    description: Optional[str] = None


class TestPlanResponse(BaseModel):
    """测试计划响应"""
    id: UUID
    project_id: UUID
    name: str
    description: Optional[str]
    created_at: datetime
    updated_at: datetime
    scenarios: List[PlanScenarioResponse] = []

    class Config:
        from_attributes = True


class TestPlanBriefResponse(BaseModel):
    """测试计划简要响应"""
    id: UUID
    project_id: UUID
    name: str
    description: Optional[str]
    created_at: datetime
    updated_at: datetime
    scenario_count: int = 0

    class Config:
        from_attributes = True


class TestPlanListResponse(BaseModel):
    """测试计划列表响应"""
    items: List[TestPlanBriefResponse]
    total: int
```

**Step 2: 创建 service.py**

实现以下服务类:

1. TestPlanService:
   - list(project_id, search, page, page_size) -> Tuple[List[TestPlan], int]
   - get(plan_id, with_scenarios=True) -> TestPlan
   - create(project_id, data) -> TestPlan
   - update(plan_id, data) -> TestPlan
   - delete(plan_id) -> None
   - duplicate(plan_id, new_name) -> TestPlan

2. PlanScenarioService:
   - list_by_plan(plan_id) -> List[PlanScenario]
   - add_scenario(plan_id, data) -> PlanScenario
   - update_scenario(plan_scenario_id, data) -> PlanScenario
   - remove_scenario(plan_scenario_id) -> None
   - reorder(plan_id, plan_scenario_ids) -> List[PlanScenario]
   - batch_add(plan_id, scenarios) -> List[PlanScenario]

**实现路由 (prefix="/projects/{project_id}/plans"):**

**计划管理:**
- GET / - 计划列表
- POST / - 创建计划
- GET /{plan_id} - 计划详情
- PUT /{plan_id} - 更新计划
- DELETE /{plan_id} - 删除计划
- POST /{plan_id}/duplicate - 复制计划

**计划-场景管理:**
- GET /{plan_id}/scenarios - 场景列表
- POST /{plan_id}/scenarios - 添加场景
- POST /{plan_id}/scenarios/batch - 批量添加场景
- PUT /scenarios/{plan_scenario_id} - 更新场景配置
- DELETE /scenarios/{plan_scenario_id} - 移除场景
- POST /{plan_id}/scenarios/reorder - 重排序场景
